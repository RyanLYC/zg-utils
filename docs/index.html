zg-utils

前端通用工具函数库

安装

 
  npm install --save zg-utils
  # or
  yarn add zg-utils

使用

 
  # 完整导入
  import ZgUtils from 'zg-utils'

  ZgUtils.getUrlParams('https://examples.com/base/path?name=Bob&age=24&sex=0')
  # or

  # 模块导入
  import { getUrlParams } from 'zg-utils'

   ZgUtils.getUrlParams('https://examples.com/base/path?name=Bob&age=24&sex=0')
